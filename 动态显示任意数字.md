#include <reg52.h>
#include <intrins.h>
#include <math.h>
#include "binary.h"
#define uint unsigned int

sbit weil = P2^0;
sbit duanl = P2^1;

unsigned char code table[]={0xfc,0x60,0xda,0xf2,0x66,0xb6,0xbe,0xe0,0xfe,0xf6,0xee,0x3e,0x9c,0x7a,0x9e,0x8e};
unsigned char code codeplace[]={B11111111,B01111111,B10111111,B11011111,B11101111,B11110111,B11111011,B11111101,B11111110};

uint wei;
uint caracter[9];
int count;


void delayms(uint ms){
   uint i,j;
   for(i=ms;i>0;i--)
      for(j=110;j>0;j--);
}

void display(){
	uint i,k;
	while(1){
		i = 1;
		k = wei+i;
		while(i <= wei){
			P1 = 0xff;
			weil = 1;
			P1 = codeplace[k];
			weil = 0;
			P1 = 0x00;
			duanl = 1;
			P1 = table[caracter[k]];
			duanl = 0;
			delayms(2);
			i++;
			k++;
		}
	}
}


void depart(){
	caracter[5]=count/1000;
	caracter[6]=count%1000/100;
	caracter[7]=count%1000%100/10; 
	caracter[8]=count%1000%100%10;
	wei = 4;
}
void main()
{	
	 count = 1234;
	 depart();
	 display();
}
